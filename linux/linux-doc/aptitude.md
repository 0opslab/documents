# To search for packages:
#要搜索包裹：
aptitude search "whatever"

# To display package records for the named package(s):
#要显示命名包的包记录：
aptitude show pkg(s)

# To install a package:
#要安装包：
aptitude install package

# To remove a package:
#要删除包裹：
aptitude remove package

# To remove unnecessary package:
#删除不必要的包：
aptitude autoclean
