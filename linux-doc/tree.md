# To display a recursive directory tree
#显示递归目录树
tree

# To make tree output contents from path `/foo/bar`
#从路径`/ foo / bar`制作树输出内容
tree /foo/bar

# To make tree omit any empty directories from the output
#使树从输出中省略任何空目录
tree --prune

# To list directories only (`-d`), and at a max depth of two levels (`-L`)
#仅列出目录（`-d`），并且最大深度为两级（`-L`）
tree -d -L 2
